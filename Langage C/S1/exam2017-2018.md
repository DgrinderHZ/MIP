	a = n;
	do{
		// Calculer le nombre de diviseurs de n
		// printf("a=%d \n", a);
		nbrDiviseur = 0;

			if(a % i == 0) // test de divisibilite

}

    	for(j=1; j <= n-i; j++)

    	if(i == 1)
