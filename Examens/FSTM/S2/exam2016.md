main(){
    int tab[100]; /// type nom;
    int i, valeur, somme, sommeM2ou5;

    somme = 0;
    i = 0;
    sommeM2ou5 = 0;

    do{
        printf("Donner une valeur : ");
        scanf("%d", &tab[i]);
        valeur = tab[i];

        if (tab[i] >= 0){
            somme = somme + tab[i];

            if ((tab[i] % 2 == 0) || (tab[i] % 5 == 0))
                sommeM2ou5 = sommeM2ou5 + tab[i];

            i = i + 1;

     }while(valeur >= 0);
