int distance(unsigned char **im, int x, int y, int u, int v, int n){
	int dist = 0, i, j;
	// on passe vers les patchs
	int xn = x-n, yn = x+n;
	int un = u-n, vn = u+n;
	int k = L*H; // le nombre de pixel
	for(i = -1; i <= 1 ; ++i)
	{
		for (j = -1; j <= 1; ++j)
		{
			dist += pow((im[xn+i][yn+j] - im[un+i][vn+j]), 2);
		}
	}
	dist *= k;
	dist = sqrt(dist);
	if(dist) dist = 1/dist;
return dist;
}
